安装express框架
npm install express

安装log4j框架
npm install log4js

安装代码自动重启库
npm install nodemon

安装mysql
npm install mysql

安装tornweb
npm install tronweb

守护进程foorever
npm install forever


//管理trx服务器
cd /data/nodejs/
forever stop ./trx/index.js 
forever start  ./trx/index.js 
forever list

//管理eth服务器
cd /data/nodejs/
forever stop ./eth/index.js 
forever start  ./eth/index.js 
forever list

//启动查看所有
forever stopall 
forever list
forever restartall
